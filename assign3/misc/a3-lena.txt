package edu.wisc.cs.sdn.vnet.rt;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.List;
import java.util.Map;
import java.util.Set;

import edu.wisc.cs.sdn.vnet.Device;
import edu.wisc.cs.sdn.vnet.DumpFile;
import edu.wisc.cs.sdn.vnet.Iface;
import net.floodlightcontroller.packet.ARP;
import net.floodlightcontroller.packet.Ethernet;
import net.floodlightcontroller.packet.IPv4;
import net.floodlightcontroller.packet.MACAddress;
import net.floodlightcontroller.packet.RIPv2;
import net.floodlightcontroller.packet.RIPv2Entry;
import net.floodlightcontroller.packet.UDP;

/**
 * @author Aaron Gember-Jacobson and Anubhavnidhi Abhashkumar
 */
public class Router extends Device
{
    /** Routing table for the router */
    private RouteTable routeTable;

    /** ARP cache for the router */
    private ArpCache arpCache;

    /** Whether RIP is currently enabled */
    private boolean ripEnabled = false;

    /** Multicast IP for RIP: 224.0.0.9 */
    private static final int RIP_MULTICAST_IP = IPv4.toIPv4Address("224.0.0.9");

    /** Broadcast MAC address */
    private static final byte[] BROADCAST_MAC = {
        (byte)0xFF, (byte)0xFF, (byte)0xFF,
        (byte)0xFF, (byte)0xFF, (byte)0xFF
    };

    /** Maximum RIP metric (infinity) */
    private static final int RIP_INFINITY = 16;

    /** Route timeout in milliseconds (30 seconds) */
    private static final long ROUTE_TIMEOUT_MS = 30000;

    /**
     * Tracks the last time each RIP-learned route was updated.
     * Key: subnet address (int), Value: System.currentTimeMillis()
     */
    private final Map<Integer, Long> routeLastUpdated = new HashMap<>();

    /**
     * Set of subnets that are directly connected (never timed out).
     */
    private final Set<Integer> directSubnets = new HashSet<>();

    /**
     * Creates a router for a specific host.
     * @param host hostname for the router
     */
    public Router(String host, DumpFile logfile)
    {
        super(host, logfile);
        this.routeTable = new RouteTable();
        this.arpCache = new ArpCache();
    }

    /** @return routing table for the router */
    public RouteTable getRouteTable()
    { return this.routeTable; }

    /**
     * Load a new routing table from a file.
     */
    public void loadRouteTable(String routeTableFile)
    {
        if (!routeTable.load(routeTableFile, this))
        {
            System.err.println("Error setting up routing table from file "
                    + routeTableFile);
            System.exit(1);
        }
        System.out.println("Loaded static route table");
        System.out.println("-------------------------------------------------");
        System.out.print(this.routeTable.toString());
        System.out.println("-------------------------------------------------");
    }

    /**
     * Load a new ARP cache from a file.
     */
    public void loadArpCache(String arpCacheFile)
    {
        if (!arpCache.load(arpCacheFile))
        {
            System.err.println("Error setting up ARP cache from file "
                    + arpCacheFile);
            System.exit(1);
        }
        System.out.println("Loaded static ARP cache");
        System.out.println("----------------------------------");
        System.out.print(this.arpCache.toString());
        System.out.println("----------------------------------");
    }

    /**
     * Start RIP: called from Main.java (or here) when no static route table
     * is provided.
     */
    public void startRIP()
    {
        this.ripEnabled = true;
        System.out.println("Starting RIP...");

        // Add directly-connected subnet routes (metric = 1, no gateway)
        for (Iface iface : this.interfaces.values())
        {
            int subnet = iface.getIpAddress() & iface.getSubnetMask();
            this.routeTable.insert(subnet, 0, iface.getSubnetMask(), iface, 1);
            directSubnets.add(subnet);
            routeLastUpdated.put(subnet, Long.MAX_VALUE); // never expire
        }

        // Send initial RIP request out all interfaces
        sendRIPRequestAll();

        // Periodic unsolicited response every 10 seconds
        new Thread(() -> {
            while (true)
            {
                try { Thread.sleep(10000); } catch (Exception e) {}
                sendRIPResponseAll();
                expireRoutes();
            }
        }).start();
    }

    /**
     * Remove route table entries that haven't been updated in 30+ seconds,
     * but never remove directly-connected subnet entries.
     */
    private void expireRoutes()
    {
        long now = System.currentTimeMillis();
        synchronized (routeLastUpdated)
        {
            Iterator<Map.Entry<Integer, Long>> it =
                    routeLastUpdated.entrySet().iterator();
            while (it.hasNext())
            {
                Map.Entry<Integer, Long> e = it.next();
                int subnet = e.getKey();
                long lastSeen = e.getValue();

                if (directSubnets.contains(subnet))
                    continue; // never expire direct routes

                if ((now - lastSeen) > ROUTE_TIMEOUT_MS)
                {
                    System.out.println("RIP: expiring route for subnet "
                            + IPv4.fromIPv4Address(subnet));
                    this.routeTable.remove(subnet, 0);
                    it.remove();
                }
            }
        }
    }

    /**
     * Handle an Ethernet packet received on a specific interface.
     */
    public void handlePacket(Ethernet etherPacket, Iface inIface)
    {
        System.out.println("*** -> Received packet: " +
                etherPacket.toString().replace("\n", "\n\t"));

if (etherPacket.getEtherType() == Ethernet.TYPE_ARP)
{
    ARP arpPacket = (ARP) etherPacket.getPayload();

    /* Learn from any ARP packet we receive */
    int senderIp = IPv4.toIPv4Address(arpPacket.getSenderProtocolAddress());
    MACAddress senderMac = MACAddress.valueOf(arpPacket.getSenderHardwareAddress());
    this.arpCache.insert(senderMac, senderIp);

    /* If this is an ARP request for one of our interfaces, reply */
    if (arpPacket.getOpCode() == ARP.OP_REQUEST)
    {
        int targetIp = IPv4.toIPv4Address(arpPacket.getTargetProtocolAddress());

        for (Iface iface : this.interfaces.values())
        {
            if (iface.getIpAddress() == targetIp)
            {
                Ethernet ether = new Ethernet();
                ether.setEtherType(Ethernet.TYPE_ARP);
                ether.setSourceMACAddress(iface.getMacAddress().toBytes());
                ether.setDestinationMACAddress(etherPacket.getSourceMACAddress());

                ARP arpReply = new ARP();
                arpReply.setHardwareType(ARP.HW_TYPE_ETHERNET);
                arpReply.setProtocolType(ARP.PROTO_TYPE_IP);
                arpReply.setHardwareAddressLength(
                        (byte) Ethernet.DATALAYER_ADDRESS_LENGTH);
                arpReply.setProtocolAddressLength((byte) 4);
                arpReply.setOpCode(ARP.OP_REPLY);

                arpReply.setSenderHardwareAddress(iface.getMacAddress().toBytes());
                arpReply.setSenderProtocolAddress(iface.getIpAddress());
                arpReply.setTargetHardwareAddress(arpPacket.getSenderHardwareAddress());
                arpReply.setTargetProtocolAddress(arpPacket.getSenderProtocolAddress());

                ether.setPayload(arpReply);

                sendPacket(ether, iface);
                return;
            }
        }
    }

    return;
}
        if (etherPacket.getEtherType() != Ethernet.TYPE_IPv4) {
            return;
        }

        if (!(etherPacket.getPayload() instanceof IPv4)) {
            return;
        }

        IPv4 ipPacket = (IPv4) etherPacket.getPayload();

        // Check for RIP (UDP port 520)
        if (ipPacket.getProtocol() == IPv4.PROTOCOL_UDP)
        {
            UDP udpPacket = (UDP) ipPacket.getPayload();
            if (udpPacket.getDestinationPort() == UDP.RIP_PORT)
            {
                handleRIPPacket(etherPacket, inIface);
                return;
            }
        }

        // Validate checksum
        short originalChecksum = ipPacket.getChecksum();
        ipPacket.resetChecksum();
        ipPacket.serialize();
        if (originalChecksum != ipPacket.getChecksum()) {
            return;
        }

        // Decrement TTL
        byte ttl = ipPacket.getTtl();
        ttl--;
        if (ttl <= 0) {
            return;
        }
        ipPacket.setTtl(ttl);
        ipPacket.resetChecksum();
        ipPacket.serialize();

        // Drop if destined for one of this router's own IPs
        for (Iface iface : this.interfaces.values())
        {
            if (ipPacket.getDestinationAddress() == iface.getIpAddress())
                return;
        }

        // Look up route
        int dstIp = ipPacket.getDestinationAddress();
        RouteEntry routeEntry = this.routeTable.lookup(dstIp);
        if (routeEntry == null) {
            System.out.println("DEBUG: No route for " + IPv4.fromIPv4Address(dstIp));

            return;
        }
        // No-route-back: never forward out the same interface it arrived on
        if (routeEntry.getInterface() == inIface) {
            return;
    }
        Iface outIface = routeEntry.getInterface();
        if (outIface == null) {
            return;
        }
        outIface = this.interfaces.get(outIface.getName());
        if (outIface == null) {
            return;
        }

        int nextHop = routeEntry.getGatewayAddress();
        if (nextHop == 0) {
            nextHop = dstIp;
        }
        ArpEntry arpEntry = this.arpCache.lookup(nextHop);
        if (arpEntry == null) {
            System.out.println("DEBUG: No ARP entry for " + IPv4.fromIPv4Address(nextHop));
            return;
        }
        if (outIface.getMacAddress() == null || arpEntry.getMac() == null) {
            return;
        }

        etherPacket.setSourceMACAddress(outIface.getMacAddress().toBytes());
        etherPacket.setDestinationMACAddress(arpEntry.getMac().toBytes());
        sendPacket(etherPacket, outIface);
    }

    // -----------------------------------------------------------------------
    // RIP packet handling
    // -----------------------------------------------------------------------

    /**
     * Handle an incoming RIP packet (request or response).
     */
    private void handleRIPPacket(Ethernet etherPacket, Iface inIface)
    {
        IPv4 ipPacket   = (IPv4) etherPacket.getPayload();
        UDP  udpPacket  = (UDP)  ipPacket.getPayload();
        RIPv2 ripPacket = (RIPv2) udpPacket.getPayload();

        if (ripPacket.getCommand() == RIPv2.COMMAND_REQUEST)
        {
            // Respond unicast back to the requester
            sendRIPResponse(inIface,
                    ipPacket.getSourceAddress(),
                    etherPacket.getSourceMACAddress());
        }
        else if (ripPacket.getCommand() == RIPv2.COMMAND_RESPONSE)
        {
            // Update route table based on received entries
            for (RIPv2Entry entry : ripPacket.getEntries())
            {
                int subnet   = entry.getAddress() & entry.getSubnetMask();
                int mask     = entry.getSubnetMask();
                int newCost  = entry.getMetric() + 1;
                if (newCost >= RIP_INFINITY) {
                    continue;
                }
                synchronized (routeLastUpdated)
                {
                    RouteEntry existing = this.routeTable.lookup(subnet);

                    if (existing == null)
                    {
                        this.routeTable.insert(
                                subnet,
                                ipPacket.getSourceAddress(),
                                mask,
                                inIface,
                                newCost);
                        routeLastUpdated.put(subnet, System.currentTimeMillis());
                        System.out.println("RIP: learned new route to "
                                + IPv4.fromIPv4Address(subnet)
                                + " cost=" + newCost);
                    }
                    else if (!directSubnets.contains(subnet))
                    {
                        if (newCost < existing.getMetric()
                                || existing.getGatewayAddress()
                                   == ipPacket.getSourceAddress())
                        {
this.routeTable.update(
        subnet,
        mask,
        ipPacket.getSourceAddress(),
        inIface,
        newCost);

                            routeLastUpdated.put(subnet,
                                    System.currentTimeMillis());
                            System.out.println("RIP: updated route to "
                                    + IPv4.fromIPv4Address(subnet)
                                    + " cost=" + newCost);
                        }
                    }
                }
            }
        }
    }

    // -----------------------------------------------------------------------
    // RIP packet sending
    // -----------------------------------------------------------------------

    /**
     * Send a RIP request out ALL interfaces (multicast).
     */
    private void sendRIPRequestAll()
    {
        System.out.println("Sending RIP request on all interfaces...");
        for (Iface iface : this.interfaces.values())
        {
            sendRIPRequest(iface);
        }
    }

    /**
     * Send an unsolicited RIP response out ALL interfaces (multicast).
     */
    private void sendRIPResponseAll()
    {
        System.out.println("Sending unsolicited RIP response on all interfaces...");
        for (Iface iface : this.interfaces.values())
        {
            sendRIPResponse(iface, RIP_MULTICAST_IP, BROADCAST_MAC);
        }
    }

private void sendRIPRequest(Iface outIface)
{
    RIPv2 rip = new RIPv2();
    rip.setCommand(RIPv2.COMMAND_REQUEST);

    RIPv2Entry entry = new RIPv2Entry();
    entry.setAddress(0);
    entry.setSubnetMask(0);
    entry.setMetric(16);
    rip.getEntries().add(entry);

    Ethernet eth = buildRIPEthernet(rip, outIface,
            RIP_MULTICAST_IP, BROADCAST_MAC);

             if(eth != null){
                sendPacket(eth, outIface);
            }
    // sendPacket(eth, outIface);
}
    /**
     * Build and send a RIP response out a single interface.
     *
     * @param outIface   interface to send on
     * @param dstIp      destination IP (multicast or unicast)
     * @param dstMac     destination MAC (broadcast or unicast)
     */
    private void sendRIPResponse(Iface outIface, int dstIp, byte[] dstMac)
    {
        RIPv2 rip = new RIPv2();
        rip.setCommand(RIPv2.COMMAND_RESPONSE);

        List<RIPv2Entry> entries = new ArrayList<>();
        for (RouteEntry re : this.routeTable.getEntries())
        {
            RIPv2Entry entry = new RIPv2Entry();
            entry.setAddress(re.getDestinationAddress());
            entry.setSubnetMask(re.getMaskAddress());
            int metric = Math.min(re.getMetric(), RIP_INFINITY);
            entry.setMetric(metric);
            entries.add(entry);
        }
        rip.setEntries(entries);

        Ethernet eth = buildRIPEthernet(rip, outIface, dstIp, dstMac);
if (eth != null) {
    sendPacket(eth, outIface);
}
        // sendPacket(eth, outIface);
    }

    /**
     * Construct a full Ethernet / IP / UDP / RIPv2 packet.
     */
    private Ethernet buildRIPEthernet(RIPv2 rip, Iface outIface,
                                      int dstIp, byte[] dstMac)
    {
        // UDP
        UDP udp = new UDP();
        udp.setSourcePort(UDP.RIP_PORT);
        udp.setDestinationPort(UDP.RIP_PORT);
        udp.setPayload(rip);

        // IPv4
        IPv4 ip = new IPv4();
        ip.setVersion((byte) 4);
        ip.setTtl((byte) 64);
        ip.setProtocol(IPv4.PROTOCOL_UDP);
        ip.setSourceAddress(outIface.getIpAddress());
        ip.setDestinationAddress(dstIp);
        ip.setPayload(udp);

        // Ethernet
        Ethernet eth = new Ethernet();
        eth.setEtherType(Ethernet.TYPE_IPv4);

        if (outIface.getMacAddress() == null) {
            System.out.println("RIP: skipping interface " + outIface.getName() + " (no MAC)");
            return null;
        }

        eth.setSourceMACAddress(outIface.getMacAddress().toBytes());
        eth.setDestinationMACAddress(dstMac);
        eth.setPayload(ip);

        return eth;
    }
}
